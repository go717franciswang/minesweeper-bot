location = find().getTarget()
box_patterns = {
        '0': Pattern().exact(),
        '1': Pattern().exact(),
        '2': Pattern().exact(),
        '3': Pattern().exact(),
        '4': Pattern().exact(),
        '5': Pattern().exact(),
        '6': Pattern().exact(),
        '7': Pattern().exact()}
x = location.getX()-448
y = location.getY()+30
box_size = 30
w = 30*box_size
h = 16*box_size

board = ""

def match(region, pattern):
    try:
        return region.findAll(pattern)
    except FindFailed:
        return []

board = ("_"*30+"\n")*16
r = Region(x, y, w, h)
for k, pattern in box_patterns.iteritems():
    for m in match(r, pattern):
        x0 = (m.getX()-x)//30
        y0 = (m.getY()-y)//30
        i = y0*31+x0
        board = board[:i] + k + board[i+1:]

print board
